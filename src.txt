import java.util.Collections;


public class NewSingleton {
    private static NewSingleton ourInstance = new NewSingleton();

    public static NewSingleton getInstance() {
        return ourInstance;
    }

    private NewSingleton() {
    }

    public static int[] toArrayInt(long input) {

        char[] string = String.valueOf(input).toCharArray();
        int size = string.length;
        int[] array = new int[size];
        for (int i = 0; i < size; i ++) {
            array[i]  = Character.getNumericValue(string[i]);
        }

        return array;
    }

    public static int[] sortArrayToNextNumber(int[] input) {

        int pointStop = 0;

        for (int i = input.length - 1; i > 0; i--) {
            if (input[i-1] < input[i]) {
                int tmp = input[i];
                input[i] = input[i-1];
                input[i-1] = tmp;
                break;
            }
        }

        for (int i = input.length - 1; i > pointStop; i--) {
            if (input[i-1] > input[i]) {
                int tmp = input[i];
                input[i] = input[i-1];
                input[i-1] = tmp;
                break;
            }
        }

        return input;
    }

    public static long arrayToNumber(int[] input) {
        String s = "";
        for(int i = 0; i < input.length; i++) {
            s += "" + input[i];
        }

        return Long.parseLong(s);
    }

    public static void main(String[] args) {
        long input = 1232333421;
        int[] a = toArrayInt(input);
        a = sortArrayToNextNumber(a);
        long value = arrayToNumber(a);
        System.out.println(input);
        System.out.println(value);

    }
}
